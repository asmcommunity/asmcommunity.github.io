I got this macro, I dont remember where I got it from... 



L MACRO Text

	LOCAL buffer, Name

	buffer textequ <>

	

	FORC chr, <Text>

		IFDIF <chr>, <">

			buffer CatStr buffer, <">, <chr> , <",>

		ENDIF

	ENDM

	buffer SUBSTR buffer, 1, @SizeStr(%buffer)-1



	.data

		align 4

		Name dw buffer,0

	.code

		EXITM <OFFSET Name>

ENDM
