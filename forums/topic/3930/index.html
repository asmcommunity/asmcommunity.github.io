reverse string again, but ....
esi - string pointer

edi - string end pointer - 4



@@:

mov eax, [esi]

mov edx, [edi]

add esi, 4

sub edi, 4

bswap eax

bswap edx

mov [edi + 4], eax

mov [esi - 4], edx



cmp edi, esi

jge @B
