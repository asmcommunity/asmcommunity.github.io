how to call DBGPrint in ASM
Get the complete source in DbgPrint.VXD
and look the file DBG.H 



_asm

	{

	lea	eax,(pszfmt + 4)

	push	eax

	push	pszfmt

	_emit	0xCD;		// int 20

	_emit	0x20

	_emit	0x2D;		// function 0x2D010100 (_Debug_Printf_Service)

	_emit	0x01

	_emit	0x01

	_emit	0x00

	add	esp, 2*4

	}
