try this
extern MessageBoxA
extern ExitProcess
extern FindWindowA
extern SendMessageA
extern ShellExecuteA
%include '\lab\vasm\inc\win32n.inc'
%include '\lab\vasm\inc\nagoa.inc'



ClassName db '#32770',0 ; <--messagebox?s



..start:
enter 0,0

call FindWindowA,ClassName,0
call SendMessageA,eax,WM_CLOSE,0,0

exit:
call ExitProcess ,0
xor eax,eax
leave
ret 16
