Re: Muliply
mov ecx, 98765432h  ;2557891634
mov eax, 7
mul ecx  ; multiplies ecx by eax

result is 42B3C4D5Eh in held in the register pair edx:eax
which are the high and low 32 bit parts of a 64 bit result
edx = 4
eax = 2B3C4D5Eh
