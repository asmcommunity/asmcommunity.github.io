Packed BCD to binary:
	mov	edx, eax

	shr	edx, 4

	and	edx, 0F0F0F0Fh

	imul	edx, edx, 10h - 10

	sub	eax, edx



	mov	edx, eax

	shr	eax, 8

	and	eax, 0FF00FFh

	imul	eax, eax, 100h - 100

	sub	edx, eax



	mov	eax, edx

	shr	edx, 16

	imul	edx, edx, 10000h - 10000

	sub	eax, edx
Similar to:
	mov	ebx, eax

	mov	ecx, 8

	xor	eax, eax

@@:	rol	ebx, 4

	mov	edx, ebx

	and	edx, 0Fh

	lea	eax, [eax*4 + eax]

	lea	eax, [eax*2 + edx]

	dec	ecx

	jne	@B
