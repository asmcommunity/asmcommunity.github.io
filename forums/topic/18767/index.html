help by converting an byte to an 2 digit hex
Here's the reverse of what you want: 



HexToInt PROC USES ebx esi pHexString:DWORD

  mov       esi, pHexString

  xor       eax, eax

  xor       ebx, ebx

  mov       bl, [esi]

  .WHILE bl != NULL

    .IF bl > '9'

      sub   bl, 7

    .ENDIF

    sub     bl, '0'

    cmp     bl, 0Fh

    ja      InvalidHexValue

    or      al, bl

    shl     eax, 4

InvalidHexValue:

    inc     esi

    mov     bl, [esi]

  .ENDW



  shr       eax, 4

  ret

HexToInt ENDP


It's not the best code in the world, but you can probably figure out how to rewrite it to do what you want.
