take a look at this sample file....

Scroll PROC
call GetCurrentThread
push 0
push eax
call SetThreadPriority

mov ebx,xpos
wsk01:

push 1 ;repaint
push height ;height
push width_ ;width
push ypos ;y ^^^^^^^^^
push ebx ;x >>>>>>>>>
push hbitmapa ;uchwyt elementu
call MoveWindow

push time
call Sleep

cmp ebx,-175
jne right

mov ebx,340
jmp wsk01

right:
dec ebx

jmp wsk01

xpos equ 63
ypos equ 20


height equ 54
width_ equ 168
time equ 32

jmp wsk01
