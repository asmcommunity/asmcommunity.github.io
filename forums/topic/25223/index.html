Re: How to convert IP address (dec to hex)
Plug it in and move on  :)
.code
; IN:  pointer to string "x.y.z.w"
; OUT: eax holds IP dword
IPStrToDw proc uses esi, pIPString:PTR BYTE
	mov esi,pIPString
	xor eax,eax
	xor ecx,ecx
	jmp @F
	.repeat
		.if edx == '.'
			or ecx,eax
			shl ecx,8
			xor eax,eax
		.else
			lea eax,
			lea eax,
		.endif
	@@:
		movzx edx,byte ptr 
		inc esi
	.until !edx
	or eax,ecx
	ret
IPStrToDw endp

;;;;;;;;;;;;;;;;;;;; Test
.data
ipString db "202.28.1.80",0
.code
start:
invoke IPStrToDw,addr ipString
