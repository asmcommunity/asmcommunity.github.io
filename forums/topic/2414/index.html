How to : get actual Time
.386
.MODEL FLAT,STDCALL
locals
jumps

UNICODE=0

include w32.inc

GetTimeFormatA PROTO :DWORD,:DWORD,:DWORD,:DWORD,:DWORD,:DWORD
GetTimeFormat equ <GetTimeFormatA>

TIME_FORCE24HOURFORMAT equ 8h

.DATA

;[--------==========================================================--------];
title1 db 'win time by Max', 0
timebuffer db "99:99:99",0
;[--------==========================================================--------];
.CODE


main:
Get_WINTIME:


call GetTimeFormat,NULL,TIME_FORCE24HOURFORMAT ,0,0,offset timebuffer,9
call MessageBox,0,offset timebuffer, offset title1, 0
call ExitProcess,0

ends
end main
