Hi Delight 



;AL contains byte

xor		ah,ah

ror		ax,4

shr		ah,4

cmp		al,10

sbb		al,69h

das

ror		eax,8

cmp		al,10

sbb		al,69h

das

rol		eax,8

;AX contains ascii hex, ready to store in a string buffer
