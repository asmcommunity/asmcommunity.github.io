jump table




Jump_table dd offset Routine_1

           dd offset Routine_2

           dd offset Routine_3	;etc.....



routine_num	dd 0		



	lea	esi,Jump_table

	mov	eax,routine_num

; then call or jump

	call	dword ptr[esi+(eax*4)]

;	jmp	dword ptr[esi+(eax*4)]





Routine_1:

	; your code

	ret

Routine_2:

	; your code

	ret

Routine_3:

	; your code

	ret
