substrings


lea esi,buffer1

lea edi,buffer2

xor ecx,ecx

@@:

mov al,BYTE PTR[esi+ecx]

inc ecx

cmp al,"<"

jne @B

add esi,ecx

xor ecx,ecx

@@:

mov al,BYTE PTR[esi+ecx]

cmp al,">"

je @F

mov BYTE PTR[edi+ecx],al

inc ecx

jmp@B

@@:
