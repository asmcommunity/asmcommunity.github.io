Hi AceEmbler

Change the following in masm.ini 



[CodeMacro]

;KeyWord,InsertBeforeActiveLine,InsertAfterActiveLine

;{I}=Copy indent from keyword line

;{C}=Cr

;{S}=Space

;{T}=Tab

;{$}=Copy Proc/Struct Name

;The macros are set up to assume AutoIndent on

1=.if,{T},{C}{I}.endif

2=.else,{T},

3=.elseif,{T},

4=.while,{T},{C}{I}.endw

5=$ macro,{T},{C}{I}endm

6=$ proc,{T}{C}{I}{T},{C}{I}{T}ret{C}{C}{I}{$} endp

7=$ struct,{T},{C}{I}{$} ends

8=$ struc,{T},{C}{I}{$} ends

{Code}

Code=$ endp,$ proc

Const={C},$ equ

Data={C},$ db,$ dw,$ dd,$ dq,$ df,$ dt,$ byte,$ word,$ dword,$ qword,$ <?>,$ {?},$ <>

Macro=endm,$ macro

Struct=$ ends,$ struct,$ struc

[CodeBlock]

1=$ proc,$ endp,,,6

2=.if,.endif,.elseif,.else,0

3=.while,.endw,.break,,0

4=$ struct,$ ends,,,6

5=$ struc,$ ends,,,6

6=$ macro,endm,,,14

7=.data,,,,16

8=.data?,,,,16

9=.const,,,,16

10=.code,,,,16

Note that {Code} should be squear brackets.
