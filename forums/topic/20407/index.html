ai vai seu algo pra tirar raiz : 


	Raiz proc Valor:DWORD

	LOCAL Resultado:DWORD

	mov ecx,Valor

	

	.while ecx

		push ecx

		mov ebx,ecx

		mov edx,ecx

		dec ebx

		.while ebx			

			add ecx,edx

			.if ecx == Valor

				mov Resultado,edx

			.endif

			dec ebx

		.endw

		pop ecx

		.break .if ecx == 0

		dec ecx

	.endw

	mov eax,Resultado

	ret

	Raiz endp
