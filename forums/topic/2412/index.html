How to : Soundcard not/found
.386
.MODEL FLAT,STDCALL


UNICODE=0

include w32.inc


extrn lstrcat : proc
extrn lstrcpy :Proc
extrn lstrcatA :Proc
extrn lstrcmpi :Proc

extrn auxGetNumDevs :Proc


.DATA

;[--------==========================================================--------];
title1 db 'detect soundcard by Max', 0
found db 'soundcard found', 0
nofound db 'soundcard not found', 0
;[--------==========================================================--------];

.CODE
main:

call GetDevices



nodevs:
call ExitProcess ; Exit the main process


GetDevices proc

call auxGetNumDevs
.if eax > 0
call MessageBox, 0, offset found, offset title1, 0
.else
call MessageBox, 0, offset nofound, offset title1, 0
.endif

ret
GetDevices endp


ends
end main
