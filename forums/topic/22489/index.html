align 16
SetCustomCursor proc
	;---load custom cursor---
	invoke LoadCursor,hInstance,1
	.if eax!=0
		invoke SetCursor,eax
		push 1
		pop eax
	.else
		xor eax,eax	
	.endif	
	ret
