BOOL CALLBACK KillScreenSaverFunc (HWND hwnd, LPARAM lParam)
{
   char szClass [32];
   GetClassName (hwnd, szClass, sizeof(szClass)-1);
   if (strcmpi (szClass, "WindowsScreenSaverClass") == 0)
      PostMessage (hwnd, WM_CLOSE, 0, 0);
   return TRUE;
}
