Maybe, this proc?

proc dec2bin,pstrdec
enter
mov ecx,
xor eax,eax ; result register
cdq ; loadable register
.loop:
lea eax,
lea eax, ; eax <- (eax*10+dl)
mov dl,
inc ecx ; dl <- (-'0')
sub dl,'0' ; ANSI to bin
cmp dl,9
jbe .loop ; continue if 0<dl<9
return eax
