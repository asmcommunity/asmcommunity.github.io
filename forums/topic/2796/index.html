Yes i have used rudeboy?s md5 library (in masm) in the past with no problems. 

MD5Sum STRUCT

	dwSum1	dd 0

	dwSum2	dd 0

	dwSum3	dd 0

	dwSum4	dd 0

MD5Sum ENDS    



MD5_Hash	MD5Sum <>



MDxInit proc uses esi MD5Ptr:DWORD

	assume esi:ptr MD5Sum

	mov esi, MD5Ptr

	mov eax, 067452301h

	mov [esi].dwSum1, eax

	mov eax, 0efcdab89h

	mov [esi].dwSum2, eax

	mov eax, 098badcfeh

	mov [esi].dwSum3, eax

	mov eax, 010325476h

	mov [esi].dwSum4, eax

	ret

MDxInit endp



invoke MDxInit, addr MD5_Hash
