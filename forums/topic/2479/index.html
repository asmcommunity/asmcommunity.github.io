hi dxantos


maybe this macro can help you (see the attachement).
this macro work actualy with NASM compiler.
if you want to work with MASM try to replace Nasm Contants with this ones :

Public Const NASMPath = "C:\masm32\bin\ml.exe"

For debug version :
Public Const NASMSwitches = "ml -nologo -c -coff -Zi -Id:\masm32\include -Fl$(OutDir)\$(InputName) -Fo$(OutDir)\$(InputName) $(InputName).asm"

For release version :
Public Const NASMSwitches = "ml -nologo -c -coff -Id:\masm32\include -Fl$(OutDir)\$(InputName) -Fo$(OutDir)\$(InputName) $(InputName).asm"
