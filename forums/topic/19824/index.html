IFDEF multiple symbols
IFDEF _WIN32_WINNT

  IF _WIN32_WINNT GT 0400h

    IF WINVER LT 0400h







    ENDIF ; WINVER LT 0400h

  ENDIF ; _WIN32_WINNT GT 0400h

ENDIF ; DEF _WIN32_WINNT
This is how I do it.
