 num:real;

function hez(x:real; n:integer):real;

var

  i:integer;

  s:real;

begin

  s:=1;

  for i:=1 to n do

    s:=s*x;

  hez:=s;

end;

function rot(num:real; m:integer):real;

var

  t,s:real;

  i,j:integer;

begin

  s:=0;

  repeat

    s:=s+1



  until hez(s,m)>trunc(num);

  s:=s-1;

  i:=0;

  t:=0;

  repeat

    inc(i);

    j:=0;

    repeat

      inc(j);

      t:=s+j/hez(10,i);

    until hez(t,m)>=num;

    j:=j-1;

    s:=s+(j/hez(10,i));

  until (i=10) or  (hez(s,m)=num);

  rot:=s;

  end;



begin

  writeln('enter number');

  readln(num);

  writeln('enter which root');

  readln(m);

  writeln('root is: ',rot(num,m):0:10);

  writeln(rot(num,m):0:10,' ^ ',m,' = ',hez(rot(num,m),m):0:5);

  readln;

end.
