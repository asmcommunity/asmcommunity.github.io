StrLenMMX proc pszString
	mov edx,;;pszString
	pxor mm1,mm1
@@:	movq mm0,
	add edx,8
	pcmpeqb mm0,mm1
	movd eax,mm0
	psrlq mm0,32
	movd ecx,mm0
	test eax,eax
	jnz @F
	test ecx,ecx
	jz @B
	emms
	bsf eax,ecx
	shr eax,3
	sub edx,;;pszString
	lea eax,
	ret 1*4
@@:	emms
	bsf eax,eax
	shr eax,3
	sub edx,;;pszString
	lea eax,
	ret	1*4
StrLenMMX endp
