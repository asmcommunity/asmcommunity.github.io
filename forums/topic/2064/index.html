You wanna hide them?

.data
;.....
Kernel32 db "kernel32.dll",0
function db "RegisterServiceProcess"
;......
.code
;...
invoke GetModuleHandle, addr Kernel32
invoke GetProcAddress, eax, addr function
mov ebx, eax ;this hides from task list
push 1 ;
push 0 ;
call ebx ;
;...
end...
