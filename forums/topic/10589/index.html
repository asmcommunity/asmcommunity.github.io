extern EndDialog

push dword DProc
push byte 0
push dword string
push byte 0
call DialogBoxParamA

call ExitProcess

ret

%define lparam ebp+20
%define wparam ebp+16
%define msg ebp+12
%define hdlg ebp+8

@@unhandled:

leave

@@wm_init:
mov eax,1
leave
ret 16
@@wm_command:
cmp dword ,IDOK
