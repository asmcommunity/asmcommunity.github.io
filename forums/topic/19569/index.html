	SendMail(

		LPTSTR	pFROM		// your email adress

	invoke SendMail,SADD ("mail.gmx.net"),			;SMPT Server

			25,					;Port (25==smpt)

			SADD ("my.email@gmx.net"),		;SMPT login name

			SADD ("my.email@gmx.net"),		;from:

			SADD ("to.email@gmx.net"),		;to:

			SADD ("I am the subject"),		;subject

	.IF eax==1
