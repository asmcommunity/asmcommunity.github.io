macro .while v1,c,v2

{

  local ..while

  __WHILE equ ..while

  local ..endw

  __ENDW equ ..endw

  __WHILE:

  if v2 eq & c eq

   cmp v1,0

   je __ENDW

  else

   cmp v1,v2

   jn#c __ELSE   <<=====? jn#c __ENDW

  end if

}
