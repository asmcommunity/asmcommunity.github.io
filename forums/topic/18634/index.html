State of Keyboard
it is stored in BIOS area 0040.

here's a code fragment how to read keys without INT 16h 



getkey:

        push    ds

        push    bx

        push    0040H

        pop     ds

        xor     ax,ax

        mov     BX,ds:[001Ah]

        cmp     BX,ds:[001Ch]

        jz      short getkey_1

        mov     AX,[BX]

        INC     BX

        INC     BX

        CMP     BX,ds:[0082h]

        JNZ     short @F

        MOV     BX,ds:[0080h]

@@:

        MOV     ds:[001Ah],BX

getkey_1:

        pop     bx

        pop     ds

        ret
