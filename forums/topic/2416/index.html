How to : create random names,diggits...
.386
.MODEL FLAT,STDCALL
locals
jumps

UNICODE=0

include w32.inc

extrn GetTickCount:proc

_wsprintfA PROCDESC WINAPI
_wsprintf TEXTEQU <_wsprintfA>

.DATA

;[--------==========================================================--------];

title1 db 'random names by Max - press OK', 0


adj_1 db "Max", 0
adj_2 db "Moritz", 0
adj_3 db "Pepe", 0
adj_4 db "Doris", 0
adj_5 db "Tina", 0
adj_6 db "Oma", 0
adj_7 db "Opa", 0
adj_8 db "Bill", 0
adj_9 db "Gates", 0

adjectives dd adj_1, adj_2, adj_3, adj_4, adj_5,adj_6, adj_7, adj_8, adj_9


.CODE

;[--------==========================================================--------];
main:

lets_do:
CALL GetTickCount
MOV BX, 9
DIV BX
call MessageBox,0,adjectives, offset title1, 1
cmp eax,1
je lets_do
call ExitProcess,0


ends
end main
