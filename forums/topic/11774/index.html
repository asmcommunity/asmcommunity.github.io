Public Function IsWindowModal(ByVal lngHwnd As Long) As Boolean



   Dim rtn As Long

   rtn = GetWindowLong(lngHwnd, GWL_EXSTYLE)

   Debug.Print rtn

   If rtn And &H40000 Then

     IsWindowModal = False

   Else

     IsWindowModal = True

   End If



End Function
