Coloring a Menu ?
This works :
LOCAL MInfo		:MENUINFO



invoke CreateSolidBrush,MyColor

mov MInfo.hbrBack,eax

mov MInfo.cbSize,SIZEOF MENUINFO

mov MInfo.fmask,MIM_BACKGROUND

invoke SetMenuInfo,hMenu,ADDR MInfo
