.data?
	Entries  dw 32768 dup (?) ; place for negatives
	Entries2 dw 32768 dup (?) ; place for positives
.code
	
Sort proc C PUBLIC ptr1,num
	local MaxCountValue:WORD,MaxCount:WORD
	.if !ptr1 || !num
		xor eax,eax
		ret
	.endif
	push edi
	push esi
	push ebx
	;-----[ init Entries ]---------[
	mov ecx,32768
	xor eax,eax
	lea edi,Entries
	rep stosd
	;------------------------------/
	
	;----[ count Entries ]---[
	mov ebx,ptr1
	mov ecx,num
	@@:
	movsx eax,word ptr
	inc Entries2
	add ebx,2
	dec ecx
	jnz @B
	;------------------------/
	
	
	;-----[ reconstruct ptr1 and get max count ]----[
	mov MaxCount,0
	mov CX,-32768
	mov esi,65536
	mov edi,ptr1
	lea ebx,Entries
	nextEntry:
		mov ax,
		.if ax
			cmp ax,MaxCount
			jle @F
				mov MaxCount,ax
				mov MaxCountValue,CX
			@@:
				mov ,CX
				add edi,2
				dec ax
			jnz @B
		.endif
		
		inc CX
		add ebx,2
		dec esi
		jnz nextEntry
	;-----------------------------------------------/
	
	pop ebx
	pop esi
	pop edi
	movsx eax,MaxCountValue
	ret
Sort endp
