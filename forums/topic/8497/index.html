How can I detect resolution of current video mode
Pseudo code:

GetDC, 0 ; desktop dc
mov hDC, eax
GetDeviceCaps, hDC, PLANES
mov nPlanes, eax
GetDeviceCaps, hDC, BITSPIXEL
mov nBpP, eax
ReleaseDC, 0, hDC
mov colordepth, nPlanes * nBpP
