You're welcome. Here is a possible solution : 



double truncar(double entrada, unsigned __int64 mask) {

  double retval;

  __asm {

    movq    MM0, mask     ;// mm0 = mask

    pandn MM0, entrada  ;// mm0 = entrada & ~ mask

    movq    retval, MM0

    emms

    fld     retval

}

}
