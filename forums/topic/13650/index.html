For C, use stdlib's system() function.
Quick example: 

[size=12]#include <stdlib.h>

#include <stdio.h>



int main( int argc, char **argv )

{

    int n;

    n = system( "copy foo.txt bar.txt" );

    if( n == -1 )

        printf( "Error #%d\n", errno );

    else

        printf( "Ok\n" );

    return 0;
