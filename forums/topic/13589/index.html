Use CtreateThread...
Pass your parameters in an array or structure that way you can have as many as you want.
.data?

ParamStruct STRUCT

param1 DWORD ?

param2 DWORD ?

param3 DWORD ?

param4 DWORD ?

ParamStruct ENDS



.code

mov eax,OFFSET ParamStruct

Invoke myProc,hWnd,uMsg,wParam,eax





myProc proc,hWnd,uMsg,wParam,lParam

mov edi,lParam

mov param1,[edi].ParamStruct.param1

mov param2,[edi].ParamStruct.param2

mov param3,[edi].ParamStruct.param3

mov param4,[edi].ParamStruct.param4
