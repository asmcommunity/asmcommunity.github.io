; Put in you data sections

.data

String   db "The quick brown fox jumps over the lazy dogs back",0

.data?

Boofer  db 50 dup(?)



[..snip..]



mov esi, OFFSET String

mov edi, OFFSET Boofer

add esi, 4                       ; Start Here

mov ecx, 10                    ; Length of SubString

rep movsb
