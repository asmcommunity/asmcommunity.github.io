StrCopyA alog.
StrCopyA alog. is for use in NASM
By Wonder Mage.:cool:

GLOBAL StrCopyA

StrCopyA:
%define OutBuffer1 ebp+8
%define InBuffer1 ebp+12
enter 0,0
jmp @StrCopyAContinue1
.data
@@iLen1: DD 0
@@iPos1: DD 0
.code
@StrCopyAContinue1:
mov ecx,dword
lea esi,
mov ecx,dword
lea edi,
@A1:
xor eax,eax
mov ecx,dword [@@iLen1]
mov al,byte
cmp al,NULL
jz @A2
mov byte ,0
inc dword [@@iLen1]
jmp @A1
@A2:
mov ecx,dword [@@iLen1]
cmp dword [@@iPos1],ecx
jge @A3
xor eax,eax
mov ecx,dword [@@iPos1]
mov al,byte
mov byte ,al
inc dword [@@iPos1]
jmp @A2
@A3:
mov eax,dword [@@iLen1]
leave
ret
