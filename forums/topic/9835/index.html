pushad

			pushf

			xor ecx,ecx

			xor eax,eax

			xor edi,edi

			mov edi,8h

			mov esi, AsmDwordNum ; point to the string, i.e: 12345678

     _start:

			mov cl,[esi]

			cmp cl,30h

			jl _lower

			cmp cl,39h

			jg _upper

			sub cl,30h

			jmp _jmp1



    _upper:

			sub cl,37h

			jmp _jmp1

	_lower:

    _jmp1:

            add eax,ecx

			cmp edi,1

			jz _out

			shl eax,4h

	_out:

			inc esi

			dec edi

			jnz _start



			mov DwordNum,eax

			popf

			popad
