How to : get Screensize from Registry
.386
.MODEL FLAT,STDCALL
locals
jumps

UNICODE=0

include w32.inc


extrn RegQueryValueExA:proc

.DATA
;--win 98 only
;[--------==========================================================--------];

title1 db 'screensize by Max', 0



SCREENSIZE_KEY DB 'Config\0001\Display\Settings\',0
sinfo db 'Resolution',0

valuename06 DB 40 dup (0)
Vdatsize06 dd 40
handle dd 0

.CODE

;[--------==========================================================--------];
main:
Get_WINTIME:
call RegOpenKeyEx,80000002h,offset SCREENSIZE_KEY,1, 1, offset handle
call RegQueryValueExA, handle, offset sinfo, 0, 0, offset valuename06, offset Vdatsize06
call RegCloseKey,handle

call MessageBox,0,offset valuename06, offset title1, 0


call ExitProcess,0



ends
end main
